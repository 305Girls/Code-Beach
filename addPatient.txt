create procedure addPatient
     @pId varchar(18),
     @name varchar(20),
     @age int,
     @gender char(2),
     @phone varchar(11),
     @dpmId varchar(4),
     @dId varchar(10),
     @qNumber int
     As
     insert
     into Patient
     values(
     @pId,@name,@age,@gender,@phone
     )
     insert
     into PatientHSP
     values(@pId,@dpmId,@dId,null,
     @qNumber,null,0,0,0
     )
